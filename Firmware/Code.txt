#include <WiFi.h>

const char* ssid = "RollingBob-RC";
const char* password = "12345678";

WiFiServer server(80);

// Direction pins
const int IN1 = 26; // D26
const int IN2 = 27; // D27
const int IN3 = 14; // D14
const int IN4 = 12; // D12

// Speed (PWM) pins
const int ENA = 25; // D25
const int ENB = 33; // D33

int speedVal = 160;

// Movement helpers
void stopMotors() {
  digitalWrite(IN1, LOW); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW); digitalWrite(IN4, LOW);
}

void forward() {
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
}

void backward() {
  digitalWrite(IN1, LOW); digitalWrite(IN2, HIGH);
  digitalWrite(IN3, LOW); digitalWrite(IN4, HIGH);
}

void leftTurn() {
  // Right motor forward, left motor stop (pivot-ish)
  digitalWrite(IN1, LOW);  digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
}

void rightTurn() {
  // Left motor forward, right motor stop (pivot-ish)
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);  digitalWrite(IN4, LOW);
}

void setup() {
  pinMode(IN1, OUTPUT); pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT); pinMode(IN4, OUTPUT);

  // PWM (ESP32 Arduino core v3.x)
  ledcAttach(ENA, 1000, 8);
  ledcAttach(ENB, 1000, 8);
  ledcWrite(ENA, speedVal);
  ledcWrite(ENB, speedVal);

  stopMotors();

  WiFi.softAP(ssid, password);
  server.begin();
}

void loop() {
  WiFiClient client = server.available();
  if (!client) return;

  String req = client.readStringUntil('\r');
  client.flush();

  // Commands
  if (req.indexOf("GET /F") != -1) forward();
  else if (req.indexOf("GET /B") != -1) backward();
  else if (req.indexOf("GET /L") != -1) leftTurn();
  else if (req.indexOf("GET /R") != -1) rightTurn();
  else if (req.indexOf("GET /S") != -1) stopMotors();

  // Speed: /SPD?SPD=0..255
  int p = req.indexOf("SPD=");
  if (p != -1) {
    int v = req.substring(p + 4).toInt();
    speedVal = constrain(v, 0, 255);
    ledcWrite(ENA, speedVal);
    ledcWrite(ENB, speedVal);
  }

  // Response
  client.print("HTTP/1.1 200 OK\r\nContent-Type: text/html\r\nConnection: close\r\n\r\n");
  client.print(HTML);
}
